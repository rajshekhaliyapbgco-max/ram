THE FIFO ACCOUNTING ENGINE (Calculates Capital Gains) // ============================================================ function runFIFOEngine() { const ss = SpreadsheetApp.openById(SS_ID); const opSheet = ss.getSheetByName('OpeningLots'); const txSheet = ss.getSheetByName('StockTransactions'); if (!opSheet || !txSheet) return; // Prevent crashing if sheets missing const opData = opSheet.getDataRange().getValues().slice(1); const txData = txSheet.getDataRange().getValues().slice(1); let virtualQueue = []; let cgtEvents = []; // A. Load Opening Lots // Assuming Columns: [LotID, ASX, Member, Date, OrigQty, RemQty, CostPerShare, TotalCost] opData.forEach(row => { if (!row[0]) return; virtualQueue.push({ lotId: String(row[0]), asx: String(row[1]).toUpperCase(), member: String(row[2]), date: new Date(row[3]), origQty: Number(row[4]), remQty: Number(row[5] !== '' ? row[5] : row[4]), costPerShare: Number(row[6]), totalCost: Number(row[7]) }); }); // B. Load Stock Transactions // Assuming Columns: [TxnID, Date, Member, ASX, Type, Qty, Price, Brokerage, Net] let transactions = []; txData.forEach(r => { if (!r[0]) return; transactions.push({ id: String(r[0]), date: new Date(r[1]), member: String(r[2]), asx: String(r[3]).toUpperCase(), type: String(r[4]).toUpperCase(), qty: Number(r[5]), net: Number(r[8]) }); }); // Sort strictly chronological transactions.sort((a, b) => a.date - b.date); // C. Execute FIFO Logic transactions.forEach(tx => { if (tx.type === 'BUY') { virtualQueue.push({ lotId: tx.id, asx: tx.asx, member: tx.member, date: tx.date, origQty: tx.qty, remQty: tx.qty, costPerShare: tx.net / tx.qty, totalCost: tx.net }); } else if (tx.type === 'SELL') { let qtyToSell = tx.qty; let totalCostBaseExtracted = 0; let consumedLots = []; let availableLots = virtualQueue.filter(l => l.asx === tx.asx && l.member === tx.member && l.remQty > 0) .sort((a, b) => a.date - b.date); for (let i = 0; i < availableLots.length; i++) { if (qtyToSell <= 0) break; let lot = availableLots[i]; let qtyTaken = Math.min(lot.remQty, qtyToSell); lot.remQty -= qtyTaken; qtyToSell -= qtyTaken; let costOfTaken = qtyTaken * lot.costPerShare; totalCostBaseExtracted += costOfTaken; consumedLots.push(lot.lotId); } let realizedGain = tx.net - totalCostBaseExtracted; // Calculate SMSF CGT Discount (1/3rd discount if held > 365 days) let daysHeld = availableLots.length > 0 ? Math.floor((tx.date - availableLots[0].date) / (1000 * 60 * 60 * 24)) : 0; let cgtEl = daysHeld >= 365 ? 'Yes' : 'No'; let discount = (cgtEl === 'Yes' && realizedGain > 0) ? realizedGain * 0.33333 : 0; let netGain = realizedGain - discount; cgtEvents.push([ tx.asx, tx.member, daysHeld, tx.date, totalCostBaseExtracted, realizedGain, cgtEl, discount, netGain, consumedLots.join(', ') ]); } }); // D. Write to 'ComputedLots' const compSheet = ss.getSheetByName('ComputedLots'); if (compSheet) { compSheet.clearContents(); let computedOutput = [['Lot ID', 'ASX', 'Member', 'Purchase Date', 'Original Qty', 'Remaining Qty', 'Cost Base Per Share', 'Total Remaining Cost', 'Days Held', 'CGT Eligible', 'Status']]; let today = new Date(); virtualQueue.forEach(lot => { if (lot.remQty > 0.01) { let days = Math.floor((today - lot.date) / (1000 * 60 * 60 * 24)); let cgtEl = days >= 365 ? 'Yes' : 'No'; computedOutput.push([ lot.lotId, lot.asx, lot.member, lot.date, lot.origQty, lot.remQty, lot.costPerShare, (lot.remQty * lot.costPerShare), days, cgtEl, 'Open' ]); } }); if (computedOutput.length > 1) { compSheet.getRange(1, 1, computedOutput.length, computedOutput[0].length).setValues(computedOutput); } else { compSheet.getRange(1, 1, 1, computedOutput[0].length).setValues([computedOutput[0]]); } } // E. Write to 'CGT_Schedule' let cgtSheet = ss.getSheetByName('CGT_Schedule'); if (!cgtSheet) cgtSheet = ss.insertSheet('CGT_Schedule'); cgtSheet.clearContents(); let cgtOutput = [['ASX', 'Member', 'Days Held', 'Sale Date', 'Cost Base', 'Raw Gain/Loss', 'CGT Elig.', 'Discount', 'Net Taxable', 'Lots Consumed']]; if (cgtEvents.length > 0) cgtOutput = cgtOutput.concat(cgtEvents); cgtSheet.getRange(1, 1, cgtOutput.length, cgtOutput[0].length).setValues(cgtOutput); } // ============================================================ // 3. DATA FETCHING (Google Sheets -> Web App) // ============================================================ function fetchAllData() { const ss = SpreadsheetApp.openById(SS_ID); // Fetch ComputedLots const lotsSheet = ss.getSheetByName('ComputedLots'); const lotsData = lotsSheet ? lotsSheet.getDataRange().getValues().slice(1) : []; const holdings = lotsData.map(r => ({ clID: r[0], asx: r[1], member: r[2], purDate: r[3], origQty: r[4], remQty: r[5], costPS: r[6], remCost: r[7], days: r[8], cgtEl: r[9], status: r[10] })).filter(h => h.remQty > 0); // Fetch Journal const journalSheet = ss.getSheetByName('Journal'); const journalData = journalSheet ? journalSheet.getDataRange().getValues().slice(1) : []; const journal = []; const tb = {}; journalData.forEach(r => { if (!r[0]) return; const dr = Number(r[7]) || 0; const cr = Number(r[8]) || 0; const memCode = `${r[9]}:${r[5]}`; journal.push({ jid: r[0], date: r[1], ref: r[2], txnType: r[3], narration: r[4], acCode: r[5], acName: r[6], debit: dr, credit: cr, member: r[9], asx: r[10] }); if (!tb[memCode]) tb[memCode] = { net: 0 }; tb[memCode].net += (dr - cr); }); // Fetch Dividends const divSheet = ss.getSheetByName('Dividends'); const divData = divSheet ? divSheet.getDataRange().getValues().slice(1) : []; const dividends = divData.map(r => ({ id: r[0], date: r[1], asx: r[2], member: r[3], total: r[6], frank: r[8], gross: r[9], method: r[10] })); return { holdings, journal, tb, dividends, smPrices: {} }; } function fetchCGT() { const ss = SpreadsheetApp.openById(SS_ID); const cgtSheet = ss.getSheetByName('CGT_Schedule'); if (!cgtSheet) return { cgt: [] }; const cgtData = cgtSheet.getDataRange().getValues().slice(1); const cgt = cgtData.map(r => ({ asx: r[0], member: r[1], days: r[2], saleDate: r[3], costBase: r[4], gainLoss: r[5], cgtEl: r[6], discount: r[7], netGain: r[8], lot: r[9] })); return { cgt }; } function triggerJournalEngine() { try { if (typeof runFromEditor === "function") runFromEditor(); } catch(e) {} }
